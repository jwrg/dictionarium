\section{Declensions}
\subsection{Nouns}

\subsubsection{First Declension}
\begin{nounchart}{Personal endings}{f}{1st}
  Nominative  & -a    & -ae \\\hline
  Genitive    & -ae   & -\=arum \\\hline
  Dative      & -ae   & -\=is \\\hline
  Accusative  & -am   & -\=os \\\hline
  Ablative    & -\=a  & -\=is \\\hline
\end{nounchart}
